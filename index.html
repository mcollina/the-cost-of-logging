Koa

 $ npm install koa-pino-logger  
var Koa = require('koa')
var logger = require('koa-pino-logger')
var app = new Koa()
app.use(logger())
app.use(async (ctx) => {
  ctx.log.info('something else')
  ctx.body = 'hello world'
})
app.listen(3000)
